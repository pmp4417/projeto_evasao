def simular_dados(n=500):
    np.random.seed(42)
    df = pd.DataFrame({
        'frequncia': np.random.normal(80, 10, n).clip(0, 100),
        'nota_mdia': np.random.normal(6, 2, n).clip(0, 10),
        'participao': np.random.randint(0, 2, n)
    })
    df['evasao'] = ((df['frequncia'] < 70) | (df['nota_mdia'] < 5)).astype(int)
    return df

def aplicar_intervencoes(df):
    df = df.copy()
    df['frequncia'] = df['frequncia'] * 1.1
    df['nota_mdia'] = df['nota_mdia'] * 1.05
    df['frequncia'] = df['frequncia'].clip(upper=100)
    df['nota_mdia'] = df['nota_mdia'].clip(upper=10)
    return df
